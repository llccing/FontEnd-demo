模板字符串对于项目中的字符串拼接修改，很有用处。

//字符串的遍历接口
for (let codePoint of 'foo') {
  console.log(codePoint);
}

// 字符遍历接口最大的优点是识别大于0xFFFF的码点，
// 传统的for循环不能识别
var text = String.fromCodePoint(0x20bb7);
for (let i = 0; i < text.length; i++) {
  console.log(text[i]);
}

for (let i of text) {
  console.log(i);
}

'x'.repeat(3);
'hello'.repeat(0);
'1'.repeat(-1);
